# Installing Tensorflow:

1) conda create -n venv pip python=3.6  # select python version

2) source activate venv

3) pip install --ignore-installed --upgrade https://storage.googleapis.com/tensorflow/linux/cpu/tensorflow-1.12.0-cp36-cp36m-linux_x86_64.whl
